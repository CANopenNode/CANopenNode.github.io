175 static inline bool_t CO_fifo_putc(CO_fifo_t *fifo, const uint8_t c) {
176  if (fifo != NULL && fifo->buf != NULL) {
177  size_t writePtrNext = fifo->writePtr + 1;
178  if (writePtrNext != fifo->readPtr &&
179  !(writePtrNext == fifo->bufSize && fifo->readPtr == 0))
180  {
181  fifo->buf[fifo->writePtr] = c;
182  fifo->writePtr = (writePtrNext == fifo->bufSize) ? 0 : writePtrNext;
183  return true;
184  }
185  }
186  return false;
187 }
