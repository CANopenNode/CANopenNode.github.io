642 static inline bool_t CO_isError(CO_EM_t *em, const uint8_t errorBit) {
643  uint8_t index = errorBit >> 3;
644  uint8_t bitmask = 1 << (errorBit & 0x7);
645 
646  return (em == NULL || index >= (CO_CONFIG_EM_ERR_STATUS_BITS_COUNT / 8)
647  || (em->errorStatusBits[index] & bitmask) != 0) ? true : false;
648 }
