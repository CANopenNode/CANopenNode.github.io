161 /* CAN bit rates
162  *
163  * CAN bit rates are initializers for array of eight CO_CANbitRateData_t
164  * objects.
165  *
166  * Macros are not used by driver itself, they may be used by application with
167  * combination with object CO_CANbitRateData_t.
168  * Application must declare following global variable depending on CO_FCY used:
169  * const CO_CANbitRateData_t CO_CANbitRateData[8] = {CO_CANbitRateDataInitializers};
170  *
171  * There are initializers for eight objects, which corresponds to following
172  * CAN bit rates (in kbps): 10, 20, 50, 125, 250, 500, 800, 1000.
173  *
174  * CO_FCY is internal instruction cycle clock frequency in kHz units. It is
175  * calculated from oscillator frequency (FOSC [in kHz]) and PLL mode:
176  * - If PLL is not used -> FCY = FOSC / 4,
177  * - If PLL x 4 is used -> FCY = FOSC,
178  * - If PLL x 16 is used -> FCY = FOSC * 4
179  *
180  * Possible values for FCY are (in three groups):
181  * - Optimal CAN bit timing on all Baud Rates: 4000, 6000, 16000, 24000.
182  * - Not so optimal CAN bit timing on all Baud Rates: 2000, 8000.
183  * - not all CANopen Baud Rates possible: 1000, 1500, 2500, 3000, 5000,
184  * 10000, 12000, 20000, 28000, 30000, 1843 (internal FRC, no PLL),
185  * 7372 (internal FRC + 4*PLL).
186  */

391 /* Structure contains timing coefficients for CAN module.
392  *
393  * CAN baud rate is calculated from following equations:
394  * FCAN = FCY * Scale - Input frequency to CAN module (MAX 30MHz for dsPIC30F)
395  * TQ = 2 * BRP / FCAN - Time Quanta
396  * BaudRate = 1 / (TQ * K) - Can bus Baud Rate
397  * K = SJW + PROP + PhSeg1 + PhSeg2 - Number of Time Quantas
398  */
399 typedef struct {
400  uint8_t scale; /* (1 or 4) Scales FCY clock - dsPIC30F specific */
401  uint8_t BRP; /* (1...64) Baud Rate Prescaler */
402  uint8_t SJW; /* (1...4) SJW time */
403  uint8_t PROP; /* (1...8) PROP time */
404  uint8_t phSeg1; /* (1...8) Phase Segment 1 time */
405  uint8_t phSeg2; /* (1...8) Phase Segment 2 time */
406  uint16_t bitrate; /* bitrate in kbps */
407 } CO_CANbitRateData_t;
