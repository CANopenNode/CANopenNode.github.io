145 /* CAN receive message structure as aligned in CAN module. */
146 /* In dsPIC33F and PIC24H this structure is used for both: transmitting and
147  * receiving to and from CAN module. (Object is ownded by CAN module).
148  */
149 typedef struct {
150  uint16_t ident; /* Standard Identifier as aligned in CAN module. 16 bits:
151  'UUUSSSSS SSSSSSRE' (U: unused; S: SID; R=SRR; E=IDE). */
152  uint16_t extIdent; /* Extended identifier, not used here */
153  uint16_t DLC :4; /* Data length code (bits 0...3) */
154  uint16_t DLCrest :12; /* Not used here (bits 4..15) */
155  uint8_t data[8]; /* 8 data bytes */
156  uint8_t dummy; /* Not used */
157  uint8_t FILHIT; /* Filter hit */
158 } CO_CANrxMsg_t;

226  *
227  * CAN bit rates are initializers for array of eight CO_CANbitRateData_t
228  * objects.
229  *
230  * Macros are not used by driver itself, they may be used by application with
231  * combination with object CO_CANbitRateData_t.
232  * Application must declare following global variable depending on CO_FCY used:
233  * const CO_CANbitRateData_t CO_CANbitRateData[8] = {CO_CANbitRateDataInitializers};
234  *
235  * There are initializers for eight objects, which corresponds to following
236  * CAN bit rates (in kbps): 10, 20, 50, 125, 250, 500, 800, 1000.
237  *
238  * CO_FCY is internal instruction cycle clock frequency in kHz units. See
239  * dsPIC33F documentation for more information on FCY.
240  *
241  * Possible values for FCY are (in three groups):
242  * - Optimal CAN bit timing on all Baud Rates: 8000, 12000, 16000, 24000.
243  * - Not so optimal CAN bit timing on all Baud Rates: 4000, 32000.
244  * - not all CANopen Baud Rates possible: 2000, 3000, 5000, 6000, 10000,
245  * 20000, 40000, 48000, 56000, 64000, 70000.

444 /* Structure contains timing coefficients for CAN module.
445  *
446  * CAN baud rate is calculated from following equations:
447  * FCAN = FCY * Scale - Input frequency to CAN module (MAX 40MHz for dsPIC33F/PIC24H and 70MHz for dsPIC33E/PIC24E)
448  * TQ = 2 * BRP / FCAN - Time Quanta
449  * BaudRate = 1 / (TQ * K) - Can bus Baud Rate
450  * K = SJW + PROP + PhSeg1 + PhSeg2 - Number of Time Quantas
451  */
452 typedef struct {
453  uint8_t scale; /* (1 or 2) Scales FCY clock - dsPIC33F and PIC24H specific */
454  uint8_t BRP; /* (1...64) Baud Rate Prescaler */
455  uint8_t SJW; /* (1...4) SJW time */
456  uint8_t PROP; /* (1...8) PROP time */
457  uint8_t phSeg1; /* (1...8) Phase Segment 1 time */
458  uint8_t phSeg2; /* (1...8) Phase Segment 2 time */
459  uint16_t bitrate; /* bitrate in kbps */
460 } CO_CANbitRateData_t;
